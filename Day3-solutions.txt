1. Selezionare tutti gli studenti iscritti al Corso di Laurea in Economia
SELECT 'students'.'name', 'students'.'surname', 'students'.'date_of_birth', 'students'.'fiscal_code', 'students'.'enrolment_date', 'students'.'registration_number', ''students'.'email'
FROM db_university.students
JOIN1 'degrees' ON 'degrees'.'id' = 'students'.'degree_id'
WHERE 'degrees'.'name' = "Corso di laurea im Economia"


2. Selezionare tutti i Corsi di Laurea Magistrale del Dipartimento di
Neuroscienze
SELECT 'degrees'.'name'
FROM 'degrees'
JOIN 'departments' ON 'departments'.'id' = "degrees'.'department_id'
WHERE 'departments'.'name' = "Dipartimento di Neuroscienze"
AND 'degrees'.'level' = "magistrale"


3. Selezionare tutti i corsi in cui insegna Fulvio Amato (id=44)
SELECT 'courses'.*
FROM 'courses'
JOIN 'course_teacher' ON 'course_teacher','course_id' = 'courses'.'id'
WHERE 'course_teacher'.'teacher_id' = 44
